using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace ConsoleApplication28
{
    class Program
    {
        static void Main(string[] args)
        {
           
            for (int i = 4; i >= 1; i--)
            {
                for (int k = 1; k <= i; k++)
                {
                 Console.Write(k);   
                }
                Console.WriteLine();
                
            }
            Console.ReadKey();
        }
    }
}
