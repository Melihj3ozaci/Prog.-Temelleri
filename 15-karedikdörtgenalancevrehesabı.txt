using System;

using System.Linq;
using System.Text;

{
    class Program
    {

        {

            sekil = Console.ReadLine();
            Console.Write("Alan hesabýiçin A Çevre hesabý için C giriniz : ");

            if (sekil=="1")
            {
                Console.Write("1.Kenarý giriniz : ");

                if (islev=="A")
                {
                    alan = kenar1 * kenar1;
                    Console.Write("Karenin alaný={0} cm2 dir", alan);
                }
                else
                {
                    cevre = 4 * kenar1;
                    Console.Write("Karenin cevresi={0} cm dir", cevre);
                }
            }
            else
            {
                Console.Write("1.Kenarý giriniz : ");
                kenar1 = Convert.ToInt16(Console.ReadLine());
                Console.Write("2.Kenarý giriniz : ");
                kenar2 = Convert.ToInt16(Console.ReadLine());
                if (islev == "A")
                {
                    alan = kenar1 * kenar2;
                    Console.Write("Dikdörtgenin alaný={0} cm2 dir", alan);
                }
                else

                    cevre = 2 * kenar1+2*kenar2;

            }
           


            Console.ReadKey();
