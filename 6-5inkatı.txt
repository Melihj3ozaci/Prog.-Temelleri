using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;

namespace ConsoleApplication34
{
    class Program
    {
        static void Main(string[] args)
        {
            int sayi;
            Console.Write("bir sayý giriniz :");
            sayi = Convert.ToInt16(Console.ReadLine());
            if (sayi % 5 == 0)
            {
                Console.Write("girdiðiniz sayý 5'in katýdýr");
            }
            else
            {
                Console.Write("girdiðiniz sayý 5'in katý deðildir");
            }

            Console.ReadKey(); 
        
        }
    }
}
